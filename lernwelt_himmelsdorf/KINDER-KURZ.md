# 🌈 Himmelsdorf

## 🎮 So spielst du

# 🐾 Tier antippen = **helfen!**
*(rechte Maustaste)*

# 🌈 Wer lebt wo? Kolibri auf der **Regenbogenbrücke**,
# ☁️ Wolkenschaf in den **Wolkenfeldern**,
# 🎈 Sturmvogel am **Ballonhafen!**

# 🗒️ Tafel antippen = **Aufgabe lösen**

# 🌦️ Wetter-Station antippen = **Wetter wechseln!**
*(Sonne → Wolken → Regen → Gewitter → Regenbogen)*

# 🌈 Regenbogen-Block setzen = **er leuchtet & wechselt die Farbe!**

# 🎨 Regenbogen-Farben bauen (rot, orange, gelb, grün, blau …)

# 🌱 Regenbogen-Samen setzen = **wächst zu einem bunten Block!**

# 🎈 Ballon setzen, antippen = **abheben!**
*(W/S fliegen, A/D lenken, Springen = hoch, Schleichen = runter)*

# 📏 Höhenmesser + Linksklick = **Wie hoch bin ich? Oben oder unten?**

# 📷 Kamera + Linksklick = **Tier-Foto fürs Logbuch**
